"Why" Questions - Understand Design Decisions

Example Questions 
Q: Why are there separate programs for Add/Inquire/Update operations?
A: This follows the IBM i best practice of separating CRUD operations:
   - Simplifies maintenance (each program has single responsibility)
   - Enables granular security (different users can have different access)
   - Reduces program size (easier to compile and debug)
   - Follows CICS transaction model (one transaction per operation)

Q: Why does LGWEBST5 exist separately from other programs?
A: LGWEBST5 is a web service adapter added later (circa 2010) to:
   - Expose existing COBOL logic via SOAP/REST APIs
   - Enable integration with external systems
   - Avoid modifying core business logic programs
   - Maintain backward compatibility with 5250 screens

Q: Why use LGCMAREA.cpy communication area?
A: LGCMAREA provides a standard way to pass data between programs:
   - Consistent interface across all programs
   - Includes return codes, error messages, transaction context
   - Enables program-to-program calls
   - Required for CICS pseudo-conversational programming

@RestController
@RequestMapping("/api/customers")
@Validated
@Slf4j
public class CustomerController {
    
    private final CustomerValidationService validationService;
    
    public CustomerController(CustomerValidationService validationService) {
        this.validationService = validationService;
    }
    
    @PostMapping("/validate")
    public ResponseEntity validateCustomer(
            @Valid @RequestBody CustomerRequest request) {
        
        log.info("Validating customer: {}", request.customerNumber());
        
        try {
            ValidationResult result = validationService.validate(request);
            
            return ResponseEntity.ok(ValidationResponse.builder()
                .customerNumber(request.customerNumber())
                .valid(result.isValid())
                .errors(result.errors())
                .timestamp(Instant.now())
                .build());
                
        } catch (ValidationException e) {
            log.error("Validation failed for customer {}: {}", 
                request.customerNumber(), e.getMessage());
            
            return ResponseEntity.badRequest()
                .body(ValidationResponse.error(e.getMessage()));
        }
    }
}

public record CustomerRequest(
    @NotBlank @Size(max = 10) String customerNumber,
    @NotBlank @Size(max = 60) String name,
    @NotNull @Past LocalDate dateOfBirth,
    @Valid Address address
) {}

public record ValidationResponse(
    String customerNumber,
    boolean valid,
    List errors,
    Instant timestamp
) {
    public static ValidationResponse error(String message) {
        return new ValidationResponse(null, false, List.of(message), Instant.now());
    }
}

@Service
@Transactional
public class CustomerValidationService {
    
    private final CustomerRepository customerRepository;
    private final ValidationRules validationRules;
    
    public ValidationResult validate(CustomerRequest request) {
        var errors = new ArrayList();
        
        // Modern validation with pattern matching
        switch (request.customerNumber().length()) {
            case 10 -> {
                if (!request.customerNumber().matches("\\d{10}")) {
                    errors.add("Customer number must be 10 digits");
                }
            }
            default -> errors.add("Customer number must be exactly 10 characters");
        }
        
        // Age validation using modern date API
        if (Period.between(request.dateOfBirth(), LocalDate.now()).getYears() < 18) {
            errors.add("Customer must be at least 18 years old");
        }
        
        // Check for duplicates
        if (customerRepository.existsByCustomerNumber(request.customerNumber())) {
            errors.add("Customer number already exists");
        }
        
        return new ValidationResult(errors.isEmpty(), errors);
    }
}

            Customer customer = customerService.findByCustomerNumber(customerNumber);
            
            CustomerResponse response = CustomerResponse.builder()
                .customerNumber(customer.getCustomerNumber())
                .name(customer.getName())
                .address(customer.getAddress())
                .dateOfBirth(customer.getDateOfBirth())
                .houseName(customer.getHouseName())
                .houseNumber(customer.getHouseNumber())
                .postcode(customer.getPostcode())
                .build();
            
            return ResponseEntity.ok(response);
            
        } catch (CustomerNotFoundException e) {
            log.warn("Customer not found: {}", customerNumber);
            return ResponseEntity.notFound().build();
            
        } catch (Exception e) {
            log.error("Database error for customer {}: {}", customerNumber, e.getMessage());
            return ResponseEntity.status(HttpStatus.INTERNAL_SERVER_ERROR)
                .body(CustomerResponse.error("Database error occurred"));
        }
    }
}

public class CustomerValidationService {

        validateDateOfBirth(request.dateOfBirth(), errors);
        
        // Address validation
        validateAddress(request.address(), errors);
        
        // Postcode validation
        validatePostcode(request.postcode(), errors);
        
        // Duplicate check
        validateUniqueness(request.customerNumber(), errors);
        
        return new ValidationResult(errors.isEmpty(), errors);
    }
    
    private void validateCustomerNumber(String customerNumber, List errors) {
        if (customerNumber == null || customerNumber.trim().isEmpty()) {
            errors.add("Customer number is required");
            return;
        }
        
        if (!customerNumber.matches("\\d{10}")) {
            errors.add("Customer number must be exactly 10 digits");
        }
    }
    
    private void validateName(String name, List errors) {
        if (name == null || name.trim().isEmpty()) {
            errors.add("Customer name is required");
            return;
        }
        
        if (name.length() > 60) {
            errors.add("Customer name cannot exceed 60 characters");
        }
        
        if (!name.matches("^[a-zA-Z\\s'-]+$")) {
            errors.add("Customer name contains invalid characters");
        }
    }
    
    private void validateDateOfBirth(LocalDate dateOfBirth, List errors) {
        if (dateOfBirth == null) {
            errors.add("Date of birth is required");
            return;
        }
        
        LocalDate today = LocalDate.now();
        int age = Period.between(dateOfBirth, today).getYears();
        
        if (age < 18) {
            errors.add("Customer must be at least 18 years old");
        }
        
        if (age > 120) {
            errors.add("Invalid date of birth - customer cannot be over 120 years old");
        }
    }
    
    private void validateAddress(String address, List errors) {
        if (address == null || address.trim().isEmpty()) {
            errors.add("Address is required");
            return;
        }
        
        if (address.length() > 160) {
            errors.add("Address cannot exceed 160 characters");
        }
    }
    
    private void validatePostcode(String postcode, List errors) {
        if (postcode == null || postcode.trim().isEmpty()) {
            errors.add("Postcode is required");
            return;
        }
        
        if (!postcodeValidator.isValid(postcode)) {
            errors.add("Invalid UK postcode format");
        }
    }
    
    private void validateUniqueness(String customerNumber, List errors) {
        if (customerRepository.existsByCustomerNumber(customerNumber)) {
            errors.add("Customer number already exists");
        }
    }
}

@Component
public class PostcodeValidator {
    
    private static final Pattern UK_POSTCODE_PATTERN = Pattern.compile(
        "^[A-Z]{1,2}[0-9R][0-9A-Z]? [0-9][ABD-HJLNP-UW-Z]{2}$"
    );
    
    public boolean isValid(String postcode) {
        if (postcode == null) {
            return false;
        }
        
        String normalized = postcode.trim().toUpperCase();
        return UK_POSTCODE_PATTERN.matcher(normalized).matches();
    }
}
