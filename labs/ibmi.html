Modern Java Code (Spring Boot) 
@Service
@Transactional
public class OrderProcessingService {
    private final OrderRepository orderRepository;
    private final CustomerRepository customerRepository;
    private final OrderLogRepository orderLogRepository;

    public OrderProcessingService(
            OrderRepository orderRepository,
            CustomerRepository customerRepository,
            OrderLogRepository orderLogRepository) {
        this.orderRepository = orderRepository;
        this.customerRepository = customerRepository;
        this.orderLogRepository = orderLogRepository;
    }

    public void processOrders() {
        orderRepository.findAll().forEach(order -> {
            Customer customer = customerRepository
                .findById(order.getCustomerId())
                .orElseThrow(() -> new CustomerNotFoundException(
                    order.getCustomerId()));

            BigDecimal discountPercent = customer.getDiscountPercent();
            BigDecimal netAmount = calculateNetAmount(
                order.getAmount(), discountPercent);

            OrderLog log = OrderLog.builder()
                .orderNumber(order.getOrderNumber())
                .customerId(order.getCustomerId())
                .netAmount(netAmount)
                .processedAt(Instant.now())
                .build();

            orderLogRepository.save(log);
        });
    }

    private BigDecimal calculateNetAmount(
            BigDecimal amount, BigDecimal discountPercent) {
        BigDecimal discount = amount
            .multiply(discountPercent)
            .divide(BigDecimal.valueOf(100), 2, RoundingMode.HALF_UP);
        return amount.subtract(discount);
    }
}

Modern Java Code (Spring Batch) 
@Component
public class InventoryUpdateProcessor 
        implements ItemProcessor {
    
    private final ReorderRepository reorderRepository;

    public InventoryUpdateProcessor(ReorderRepository reorderRepository) {
        this.reorderRepository = reorderRepository;
    }

    @Override
    public InventoryUpdate process(Inventory inventory) {
        int newQuantity = inventory.getQuantityOnHand() 
            - inventory.getQuantityOrdered();
        
        boolean needsReorder = newQuantity < inventory.getReorderPoint();
        
        if (needsReorder) {
            ReorderRequest reorder = ReorderRequest.builder()
                .itemId(inventory.getItemId())
                .currentQuantity(newQuantity)
                .reorderPoint(inventory.getReorderPoint())
                .requestedAt(Instant.now())
                .build();
            
            reorderRepository.save(reorder);
        }
        
        return InventoryUpdate.builder()
            .itemId(inventory.getItemId())
            .newQuantity(newQuantity)
            .reorderFlag(needsReorder)
            .build();
    }
}

// Java Service Layer (replaces stored procedure)
@Service
public class OrderService {
    private final OrderRepository orderRepository;

    public OrderService(OrderRepository orderRepository) {
        this.orderRepository = orderRepository;
    }

    public BigDecimal getCustomerOrderTotal(Integer customerId) {
        return orderRepository
            .findByCustomerIdAndOrderStatus(customerId, 'A')
            .stream()
            .map(Order::getOrderAmount)
            .reduce(BigDecimal.ZERO, BigDecimal::add);
    }
}

@Repository
public interface OrderRepository extends JpaRepository {
    @Query("SELECT o FROM Order o WHERE o.customerId = :customerId " +
           "AND o.orderStatus = :status")
    List findByCustomerIdAndOrderStatus(
        @Param("customerId") Integer customerId,
        @Param("status") Character status);
}

FROM maven:3.9-eclipse-temurin-21 AS build
WORKDIR /app
COPY pom.xml .
COPY src ./src
RUN mvn clean package -DskipTests

# Runtime stage
FROM eclipse-temurin:21-jre-alpine
WORKDIR /app

# Create non-root user
RUN addgroup -S appgroup && adduser -S appuser -G appgroup

# Copy JAR from build stage
COPY --from=build /app/target/*.jar app.jar

# Set ownership
RUN chown -R appuser:appgroup /app

# Switch to non-root user
USER appuser

# Health check
HEALTHCHECK --interval=30s --timeout=3s --start-period=40s \
  CMD wget --no-verbose --tries=1 --spider http://localhost:8080/actuator/health || exit 1

# Expose port
EXPOSE 8080

# Run application
ENTRYPOINT ["java", "-XX:+UseContainerSupport", "-XX:MaxRAMPercentage=75.0", "-jar", "app.jar"]
