const server = new Server(
  {
    name: "filesystem-mcp-server",
    version: "0.1.0",
  },
  {
    capabilities: {
      resources: {},
      tools: {},
    },
  }

server.setRequestHandler(ListResourcesRequestSchema, async () => {

Step 5: Client Integration

Bob demonstrates how to integrate the MCP server with various clients and AI models.

Client Example 
import mcp

# Connect to MCP server
client = mcp.Client("filesystem-mcp-server")
await client.connect()

# List available resources
resources = await client.list_resources()
for resource in resources.resources:
    print(f"Found resource: {resource.name}")

# Read a file resource
content = await client.read_resource("file:///workspace/README.md")
print(f"File content: {content.content}")

# Execute a tool (list directory)
result = await client.call_tool("list_directory", {
    "path": "/workspace"
})
print(f"Directory listing: {result.content}")
